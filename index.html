function dirSaveComp(){
  const nom=document.getElementById('dir-nom').value.trim();
  if(!nom){alert('Ingresá el nombre de la compañía.');return;}
  const contactos=[];
  document.querySelectorAll('#dir-ct-form > div').forEach(div=>{
    const n=div.querySelector('.dir-ct-n').value.trim();
    const o=div.querySelector('.dir-ct-o').value.trim();
    const m=div.querySelector('.dir-ct-m').value.trim();
    if(n||m) contactos.push({nombre:n,obs:o,mail:m});
  });
  const c={
    nombre:nom,
    cuit:document.getElementById('dir-cuit').value.trim(),
    notas:document.getElementById('dir-notas').value.trim(),
    contactos
  };
  if(dirEditIdx!==null) companias[dirEditIdx]=c;
  else companias.push(c);
  guardar();
  close2('modal-dir');
  renderDirectorio();
  renderComps();
  refreshCompList();
}
function dirDelComp(i){
  const nr=reclamos.filter(r=>r.comp===companias[i].nombre&&r.estado!=='Cerrado').length;
  const msg=nr>0
    ?'Esta compañía tiene '+nr+' reclamo(s) activo(s). ¿Eliminarla igualmente del directorio?'
    :'¿Eliminar '+companias[i].nombre+' del directorio?';
  if(confirm(msg)){companias.splice(i,1);guardar();renderDirectorio();renderComps();refreshCompList();}
}

function toggleEstadoDropdown(e){
  e.stopPropagation();
  const d=document.getElementById('estado-dropdown');
  d.style.display=d.style.display==='none'?'block':'none';
}
function toggleAllEstados(cb){
  if(cb.checked){
    document.querySelectorAll('.est-cb').forEach(c=>c.checked=false);
  }
  updateEstadoLabel();
  renderTable();
}
function onEstadoCbChange(){
  const anyChecked=Array.from(document.querySelectorAll('.est-cb')).some(cb=>cb.checked);
  document.getElementById('est-all').checked=!anyChecked;
  updateEstadoLabel();
  renderTable();
}
function updateEstadoLabel(){
  const checked=Array.from(document.querySelectorAll('.est-cb:checked')).map(cb=>cb.value);
  const lbl=document.getElementById('estado-dropdown-label');
  if(!checked.length) lbl.textContent='Todos los estados';
  else if(checked.length===1) lbl.textContent=checked[0];
  else lbl.textContent=checked.length+' estados';
}

  const sumCob=cobrados.reduce((s,r)=>s+Number(r.hon||0),0);
  const sumPend=pendientes.reduce((s,r)=>s+Number(r.hon||0),0);

}
